017package com.gmail.woodyc40.commons.reflection.chain;
018
019import com.gmail.woodyc40.commons.reflection.ConstructorManager;
020import com.gmail.woodyc40.commons.reflection.ReflectionTool;
021import com.gmail.woodyc40.commons.reflection.impl.ReflectAccess;
022import lombok.RequiredArgsConstructor;
023
024import java.lang.reflect.Constructor;
025import java.util.*;
026
027/**
028 * Implementation for {@link com.gmail.woodyc40.commons.reflection.chain.ConstructLink}. See class description of
029 * implemented interface.
030 *
031 * @author AgentTroll
032 * @version 1.0
033 * @since 1.0
034 */
035@RequiredArgsConstructor
036public class ConstructLinkImpl implements ConstructLink {
037    private final Class<?>        base;
038    private final ReflectionChain parent;
039    private final Collection<Object> params = new ArrayList<>();
040    private ConstructorManager<?> constructor;
041
042    @Override public ConstructLink last(int index) {
043        this.params.add(this.parent.returned.get(index).getClass());
044        return this;
045    }
046
047    @Override public ConstructLink param(Object... obj) {
048        Collections.addAll(this.params, obj);
049        return this;
050    }
051
052    @Override public ConstructLink.Creator creator() {
053        return new CreatorImpl();
054    }
055
056    @Override public ConstructLink construct(Class... params) {
057        this.constructor = ReflectAccess.accessConstructor(ReflectionTool.forConstruct(this.base, params));
058        return this;
059    }
060
061    @Override public ConstructLink constructFuzzy(int params, int index) {
062        List<Constructor<?>> list = new ArrayList<>();
063        for (Constructor<?> c : this.base.getDeclaredConstructors())
064            if (c.getParameterTypes().length == params)
065                list.add(c);
066        this.constructor = ReflectAccess.accessConstructor(list.get(index));
067
068        return this;
069    }
070
071    @Override public ConstructorManager<?> getManager() {
072        return this.constructor;
073    }
074
075    private class CreatorImpl implements ConstructLink.Creator {
076        @Override public ReflectionChain create() {
077            return null;
078        }
079    }
080}
